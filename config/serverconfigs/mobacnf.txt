function commit_and_push {

  git status
  #echo "Нажмите Enter, чтобы продолжить..."
  read

  # Определяем сообщение коммита
  local commit_message="${@:-updates}"

  # Выполняем коммит и пуш
  git add . && git commit -a -m "$commit_message" && git push
}
